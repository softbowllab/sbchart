Stacked Area 차트 기본 예제입니다. data.type="stackarea"로 지정합니다.

Spline(곡선) 차트로도 가능합니다. 곡선은 data.type="stackarea-spline"로 지정합니다.

Example

Code

{% highlight javascript %} sb.chart.render("#chartWrap", { data: { type: "stackarea", columns: [ ["2015",3.51,4.6,3.43,7.96,3.37,5.83,0.35,3.23,0.09,4.47,2.21,1.13,0.77,0.35,0.92,2.65,1.64,8.08], ["2016",0.71,2.14,3.18,1.84,0.79,0.35,0.28,0.44,0.79,0.84,1.33,0.7,1.51,0.09,1.05,1.66,0.58,4.63], ["2017",1.48,3.64,2.35,1.29,1.42,1.34,1.51,1.08,4.29,1.67,2.4,0.36,0.53,1.98,2.23,0.9,1.62,1.66] ] } }); {% endhighlight %}
