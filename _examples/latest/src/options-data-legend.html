범례명을 변경합니다.
따로 지정하지 않는다면 첫 번째 칼럼 값이 그대로 범례명으로 사용됩니다.

Options

  • data.legends

    data.legends = { 2015: "'15년", 2016: "'16년", 2017: "'17년" }

Example

Code

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { data: { type: "bar", columns: [ ["2015",3.51,4.6,3.43,7.96,3.37,5.83,0.35,3.23,-0.09,4.47,2.21,1.13,0.77,0.35,0.92,2.65,1.64,8.08], ["2016",0.71,2.14,3.18,-1.84,0.79,0.35,0.28,0.44,0.79,0.84,1.33,-0.7,-1.51,0.09,1.05,-1.66,-0.58,4.63], ["2017",1.48,3.64,2.35,1.29,1.42,1.34,1.51,-1.08,4.29,1.67,2.4,-0.36,-0.53,1.98,2.23,-0.9,-1.62,1.66] ] } }); document.getElementById("chkLegends").onchange = function() { if(this.checked) { chart.data({ legends: {2015:"'15년",2016:"'16년",2017:"'17년"} }).render(); } else { delete chart.data().legends; chart.render(); } } {% endhighlight %}
