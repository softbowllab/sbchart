--- title: Pie Chart date: 2018-07-26 categories: ["latest","options"] order: 423 ---

원 차트의 라벨을 숨기거나 보여주고 커스트마이징(포멧) 합니다.
영역 사이 공간을 지정합니다.

Example

extend.pie.label.show = true/false
extend.pie.label.format = function(value, ratio, id) { return value + " (" + ratio.toFixed(4)*100 + "%)"; }
extend.pie.padding = 0

Code

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { data : { type: "pie", columns: [ ["2015",3.51], ["2016",0.71], ["2017",1.48] ] }, extend: { pie: { label: { show: false }, padding: 2 } } }); // hidden/show document.getElementById("chkShow").onchange = function() { chart.extend({ pie: { label: { show:this.checked } } }); if(document.getElementById("chkFormat").checked) { chart.extend({ pie: { label: { format:function(value, ratio, id) { return value + " (" + ratio.toFixed(4)*100 + "%)"; } } } }); } else { delete chart.extend().pie.label.format; } chart.render(); }; // format document.getElementById("chkFormat").onchange = function() { document.getElementById("chkShow").onchange(); }; // padding document.getElementById("txtPadding").onchange = function() { chart.extend({ pie: { padding: this.value } }).render(); }; {% endhighlight %}
