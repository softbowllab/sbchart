차트의 최상위 엘리먼트에 CSS 클래스명을 지정합니다. Class명 통하여 스타일 변경이 가능합니다.

Example

global.svg.classname="sbchart"

Code

{% highlight css %} .sbchart {background-color: lightcyan;} .sbchart .sbchart-axis-x {font-size: 20px;} {% endhighlight %}

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { global : { svg: {classname:"sbchart"} }, data : { type: "bar", columns: [ ["온라인쇼핑몰 거래액", 7707750, 7573752, 7914913, 7386713, 8708523, 8624841, 8785755, 8048044, 9083890] ] } }); //classname document.getElementById("chkClassname").onchange = function() { if (this.checked) { chart.global().svg.classname = "sbchart" } else { delete chart.global().svg.classname; } chart.render(); }; {% endhighlight %}
