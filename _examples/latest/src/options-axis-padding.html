X, Y축의 차트 영역에 Padding을 지정합니다. 이때 단위는 tick 단위입니다.


X축
left
axis.x.padding.left=1
right
axis.x.padding.right=1
Y축
top
axis.y.padding.top=1
bottom
axis.y.padding.bottom=1

Code

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { data : { type: "bar", columns: [ ["2017",1.48,3.64,2.35,1.29,1.42,1.34,1.51,-1.08,4.29,1.67,2.4,-0.36,-0.53,1.98,2.23,-0.9,-1.62,1.66] ] }, axis: { x: { padding: { left: 1, right: 3 } }, y: { padding: { top: 1, bottom: 1 } } } }); {% endhighlight %}
