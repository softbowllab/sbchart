도넛 차트의 라벨을 숨기거나 보여주고 커스트마이징(포멧) 합니다.
영역 사이 공간을 지정합니다.
내부 원의 크기를 지정합니다.

Example

extend.donut.label.format = function(value, ratio, id) { return value + " (" + ratio.toFixed(4)*100 + "%)"; }
extend.donut.padding = 0
extend.donut.innerRadius = 0

Code

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { data : { type: "donut", columns: [ ["2015",3.51], ["2016",0.71], ["2017",1.48] ] }, extend: { donut: { label: { show: false }, innerRadius: 0, padding: 2 } } }); // hidden/show document.getElementById("chkShow").onchange = function() { chart.extend({ donut: { label: { show:this.checked } } }); if(document.getElementById("chkFormat").checked) { chart.extend({ donut: { label: { format:function(value, ratio, id) { return value + " (" + ratio.toFixed(4)*100 + "%)"; } } } }); } else { delete chart.extend().donut.label.format; } chart.render(); }; // format document.getElementById("chkFormat").onchange = function() { document.getElementById("chkShow").onchange(); }; // padding document.getElementById("txtPadding").onchange = function() { chart.extend({ donut: { padding: this.value } }).render(); }; // innerRadius document.getElementById("txtInnerRadius").onchange = function() { chart.extend({ donut: { innerRadius: this.value } }).render(); }; {% endhighlight %}
