X, Y축의 차트 영역에 Padding을 지정한다. 이때 단위는 tick 단위이다.


X축
left
axes.x.padding.left=1
right
axes.x.padding.right=1
Y축
top
axes.y.padding.top=1
bottom
axes.y.padding.bottom=1

Code

{% highlight javascript %} var conf = { global : { size : { width: 800, height: 320 } }, data : { type: "bar", columns: [ ["2017",1.48,3.64,2.35,1.29,1.42,1.34,1.51,-1.08,4.29,1.67,2.4,-0.36,-0.53,1.98,2.23,-0.9,-1.62,1.66] ] }, axes: { x: { padding: { left: 1, right: 3 } }, y: { padding: { top: 1, bottom: 1 } } } }; SBChart.render("#chartWrap", conf); {% endhighlight %}
