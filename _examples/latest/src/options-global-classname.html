차트의 최상위 엘리먼트에 CSS 클래스명을 지정합니다. Class명을 통하여 스타일 변경이 가능합니다.

Options

  • global.svg.classname

    차트에 classname을 적용합니다.

    sbchart1, sbchart2를 각각 입력해보세요.

Example

Code

{% highlight css %} .sbchart1 { background-color: lightcyan; } .sbchart1 .sbchart-axis-x { font-size: 20px; } .sbchart2 { background-color: yellow; } {% endhighlight %}

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { global: { svg: { classname: document.getElementById("txtClassname").value } }, data: { type: "bar", columns: [ ["온라인쇼핑몰 거래액", 7707750, 7573752, 7914913, 7386713, 8708523, 8624841, 8785755, 8048044, 9083890] ] } }); //classname document.getElementById("txtClassname").onchange = function() { chart.global().svg.classname = this.value; chart.render(); }; {% endhighlight %}
