3. 차트 그려주기

데이터와 옵션을 지정하여 차트를 랜더링합니다.

차트를 생성하는 방법은 아래와 같이 3가지 방법이 있습니다.

{% highlight javascript %} var chart = sb.chart.render("#chartWrap", { data : { columns: [ ["data1", 5, 2, 3, 4, 5, 6], ["data2", 15, -12, 13, -14, 15, 16], ["data3", 25, 22, 23, 24, -25, 26], ] } }); {% endhighlight %}

{% highlight javascript %} var chart = new sb.chart("#chartWrap", { data : { columns: [ ["data1", 5, 2, 3, 4, 5, 6], ["data2", 15, -12, 13, -14, 15, 16], ["data3", 25, 22, 23, 24, -25, 26], ] } }); chart.render(); {% endhighlight %}

{% highlight javascript %} var chart = new sb.chart("#chartWrap"); chart.render({ data : { columns: [ ["data1", 5, 2, 3, 4, 5, 6], ["data2", 15, -12, 13, -14, 15, 16], ["data3", 25, 22, 23, 24, -25, 26], ] } }); {% endhighlight %}
