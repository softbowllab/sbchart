산점도 (Scatter Plot) 차트 기본 예제입니다. data.type="scatter"로 지정합니다.

data.xs를 사용하여 X축의 값을 설정하여 사용합니다.
data.xs는 {y축을 표현할 범례 아이디: x축을 표현할 범례 아이디}형태로 입력하여 차트를 표현합니다.

Example

Code

{% highlight javascript %} sb.chart.render("#chartWrap", { data: { type: "scatter", xs: { data1: "data1_x", data2: "data2_x" }, columns: [ ["data1_x",3.5,3.6,3.4,3.8,3.2,3.3,3.2,3.5,3.6,2.9,3.0,3.1,3.1,3.0,3.3,3.7,2.9,3.3,3.6,3.5,3.1,3.3,3.7,3.4], ["data2_x",2.1,2.4,2.7,2.2,2.7,3.0,2.8,2.3,2.4,2.6,2.9,2.5,3.1,2.7,2.4,2.2,2.8,2.7,2.4,2.7,2.9,2.6,2.4,2.5], ["data1",1.1,1.0,1.6,1.4,1.3,1.1,1.5,1.1,1.0,1.5,1.2,1.7,0.9,1.0,0.9,1.3,1.1,0.9,1.3,1.2,1.0,1.4,1.5,1.2], ["data2",2.0,2.2,2.4,2.3,1.6,2.1,1.9,2.2,1.6,1.9,1.8,2.0,2.2,1.7,2.2,1.9,2.1,1.7,1.7,2.0,1.7,2.3,1.9,1.8] ] }, extend: { point: { r: 4 } } }); {% endhighlight %}
