그룹별로 각각 막대, 선, 곡선 차트를 같이 그려준 예제입니다.
data.types옵션으로 그룹별 차트 종류를 지정합니다.
차트 종류가 지정되지 않으면 라인 차트로 렌더링 됩니다.

Example

Code

{% highlight javascript %} sb.chart.render("#chartWrap", { data: { types: {"2015": "bar", "2016": "bar", "2017": "spline"}, columns: [ ["2015",3.51,4.6,3.43,7.96,3.37,5.83,0.35,3.23,0.09,4.47,2.21,1.13,0.77,0.35,0.92,2.65,1.64,8.08], ["2016",0.71,2.14,3.18,1.84,0.79,0.35,0.28,0.44,0.79,0.84,1.33,0.7,1.51,0.09,1.05,1.66,0.58,4.63], ["2017",1.48,3.64,2.35,1.29,1.42,1.34,1.51,1.08,4.29,1.67,2.4,0.36,0.53,1.98,2.23,0.9,1.62,1.66] ] } }); {% endhighlight %}
